# Default to (primary) SD
rootdev=mmcblk0p3

if itest.b *0x28 == 0x02 ; then
# U-Boot loaded from eMMC or secondary SD so use it for rootfs too
echo "U-boot loaded from eMMC or secondary SD"
rootdev=mmcblk1p3
fi

setenv bootargs console=${console} root=/dev/${rootdev} rootwait panic=10 init=/init androidboot.selinux=permissive androidboot.hardware=orangepi ${extra}
load mmc 0:1 ${fdt_addr_r} ${fdtfile}
load mmc 0:1 ${kernel_addr_r} uImage
bootm ${kernel_addr_r} - ${fdt_addr_r}
